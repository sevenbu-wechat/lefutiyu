#创建数据库
CREATE DATABASE IF NOT EXISTS loveSports;
#创建报名信息表
drop database lovesports;
use loveSports;
show variables like 'character_set_database';
alter database loveSports character set utf8;
create table SIGN_UP_USER(
id int auto_increment PRIMARY KEY  COMMENT '设置主键自增',
created_at TIMESTAMP NOT NULL DEFAULT NOW() COMMENT '创建时间',
created_by varchar(32) NOT NULL DEFAULT 'SYS' COMMENT '创建人',
updated_at TIMESTAMP NOT NULL DEFAULT NOW() COMMENT '更新时间',
updated_by varchar(32) NOT NULL DEFAULT 'SYS' COMMENT '更新人',
version bigint NOT NULL default 0 COMMENT '乐观锁',
is_valid varchar(1) not null default '1' COMMENT '删除位',
user_name varchar(128) NOT NULL COMMENT '用户姓名',
train_base varchar(32) NOT NULL COMMENT '训练基地',
week_day varchar(1) not null COMMENT '星期日期',
time_schedule varchar(128) not null COMMENT '时间段',
age int not null COMMENT '年龄',
height int not null COMMENT '身高',
mobile varchar(32) not null COMMENT '手机号码',
channel varchar(4) not null COMMENT '了解渠道',
remark varchar(1024) COMMENT '家长留言'
)COMMENT '报名信息表';
